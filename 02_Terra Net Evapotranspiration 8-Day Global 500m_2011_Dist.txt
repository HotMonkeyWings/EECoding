var month_mean = ee.List.sequence(0,4*12,3).map(function(n) { // .sequence: number of years from starting year to present
  var start = ee.Date('2013-01-01').advance(n, 'month'); // Starting date
  var end = start.advance(3, 'month'); // Step by each iteration

  return ee.ImageCollection("MODIS/006/MOD16A2")
        .filterDate(start, end)
        .mean()
        .select('ET')
        .set('system:time_start', start.millis());
});
print(month_mean); 

var collection = ee.ImageCollection(month_mean);
print(collection);

print(collection.size())

function getRegions(Icol, features) {
  return Icol.map(function(image){
    var means = image.reduceRegions({
      reducer: ee.Reducer.mean(),
      collection: features.select('ST_NM', 'DISTRICT'),
      scale: 10
    }).filter(ee.Filter.notNull(['mean']))
    .map(function(f) {
      return f.set('date', image.date().format("YYYY-MM-dd"))
    })
    return means
  }).flatten()
}

var state = ee.FeatureCollection("users/dev_b180297cs/final");
